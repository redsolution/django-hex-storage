h1. Install

h2. Buildout

<pre>
[buildout]
parts = ...
        django-hex-storage

[django]
extra-paths = ...
              ${django-hex-storage:location}

[django-hex-storage]
recipe = zerokspot.recipe.git
repository = ssh://verne.redsolution.ru/srv/git/django-hex-storage.git
</pre>

h2. settings.py

<pre>
INSTALLED_APPS = (
...
    'hex_storage',
)
...
DEFAULT_FILE_STORAGE = 'hex_storage.HexFileSystemStorage'
...
</pre>
